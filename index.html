POWERSHELL · Inventaire_Parc_SJDD.ps1 
# ── Export inventaire AD + WMI vers CSV ──────────────────
$computers = Get-ADComputer -Filter * -Properties *
$results   = @()

foreach ($pc in $computers) {
    $os  = Get-WmiObject Win32_OperatingSystem -ComputerName $pc.Name
    $ram = [math]::Round($os.TotalVisibleMemorySize / 1MB, 2)

    $results += [PSCustomObject]@{
        Nom           = $pc.Name
        OS            = $os.Caption
        RAM_Go        = $ram
        Etablissement = ($pc.DistinguishedName -split ',OU=')[2]
    }
}

$results | Export-Csv "C:\Rapports\Inventaire_$(Get-Date -Format 'yyyy-MM-dd').csv" -Encoding UTF8
POWERSHELL · Montage_Lecteurs.ps1 (GPO Logon Script) 
# ── Montage conditionnel selon groupe AD ─────────────────
$user   = $env:USERNAME
$groups = (New-Object DirectoryServices.DirectorySearcher(
             "(&(sAMAccountName=$user))")).FindOne().Properties['memberof']

if ($groups -match "GG_Soignants") {
    New-PSDrive -Name "S" -PSProvider FileSystem -Persist `
        -Root "\\SRV-FICHIERS\Soins"
}

if ($groups -match "GG_Direction") {
    New-PSDrive -Name "D" -PSProvider FileSystem -Persist `
        -Root "\\SRV-FICHIERS\Direction"
}

New-PSDrive -Name "P" -PSProvider FileSystem -Persist `
    -Root "\\SRV-FICHIERS\Public" # Lecteur commun
